\newcommand{\computeworkranges}{%
  \setcounter{cumpages}{0}%
  \setcounter{tocslot}{1}%
  \loop\unless\ifnum\value{tocslot}>\value{maxworkorder}%
    \ifcsname order@\arabic{tocslot}\endcsname%
      \edef\tmpkey{\csname order@\arabic{tocslot}\endcsname}%
      \stepcounter{cumpages}%
      \expandafter\xdef\csname w@\tmpkey @start\endcsname{\arabic{cumpages}}%
      \addtocounter{cumpages}{\csname w@\tmpkey @pages\endcsname}%
      \addtocounter{cumpages}{-1}%
      \expandafter\xdef\csname w@\tmpkey @end\endcsname{\arabic{cumpages}}%
      \ifnum\csname w@\tmpkey @start\endcsname=\csname w@\tmpkey @end\endcsname
        \expandafter\xdef\csname w@\tmpkey @range\endcsname{\padpagenum{\csname w@\tmpkey @start\endcsname}}%
      \else
        \expandafter\xdef\csname w@\tmpkey @range\endcsname{\padpagenum{\csname w@\tmpkey @start\endcsname}\textendash\padpagenum{\csname w@\tmpkey @end\endcsname}}%
      \fi
    \fi%
    \stepcounter{tocslot}%
  \repeat
}
